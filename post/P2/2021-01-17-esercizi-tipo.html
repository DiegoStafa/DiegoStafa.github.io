int f(ios& ref)
{
    istream i* = dynamic_cast<istream*>(&ref)
    ostream o* = dynamic_cast<ostream*>(&ref)

    if(i && o) return 0;
    if(i) return -1;
    if(o) return 1;
    return 9;
}
